3import scipp as sc 

4from typing import Union 

5from .. import choppers as ch 

6import scipp.constants as constants 

7 

8 

9def frames_analytical(data: Union[sc.DataArray, sc.Dataset]) -> sc.Dataset: 

10 """ 

11 Compute analytical frame boundaries and shifts based on chopper 

12 parameters and detector pixel positions. 

13 A set of frame boundaries is returned for each pixel. 

14 The frame shifts are the same for all pixels. 

15 See Schmakat et al. (2020); 

16 https://www.sciencedirect.com/science/article/pii/S0168900220308640 

17 for a description of the procedure. 

18 

19 TODO: This currently ignores scattering paths, only the distance from 

20 source to pixel. 

21 For imaging, this is what we want, but for scattering techniques, we should 

22 use l1 + l2 in the future. 

23 """ 

24 

25 # Identify the WFM choppers based on their `kind` property 

26 wfm_choppers = {} 

27 for name in ch.find_chopper_keys(data): 

28 chopper = data.meta[name].value 

29 if chopper["kind"].value == "wfm": 

30 wfm_choppers[name] = chopper 

31 if len(wfm_choppers) != 2: 

32 raise RuntimeError("The number of WFM choppers is expected to be 2, " 

33 "found {}".format(len(wfm_choppers))) 

34 # Find the near and far WFM choppers based on their positions relative to the source 

35 wfm_chopper_names = list(wfm_choppers.keys()) 

36 if (sc.norm(wfm_choppers[wfm_chopper_names[0]]["position"].data - 

37 data.meta["source_position"]) < 

38 sc.norm(wfm_choppers[wfm_chopper_names[1]]["position"].data - 

39 data.meta["source_position"])).value: 

40 near_index = 0 

41 far_index = 1 

42 else: 

43 near_index = 1 

44 far_index = 0 

45 near_wfm_chopper = wfm_choppers[wfm_chopper_names[near_index]] 

46 far_wfm_chopper = wfm_choppers[wfm_chopper_names[far_index]] 

47 

48 # Compute distances for each detector pixel 

49 detector_positions = data.meta["position"] - data.meta["source_position"] 

50 

51 # Container for frames information 

52 frames = sc.Dataset() 

53 

54 # Distance between WFM choppers 

55 dz_wfm = sc.norm(far_wfm_chopper["position"].data - 

56 near_wfm_chopper["position"].data) 

57 # Mid-point between WFM choppers 

58 z_wfm = 0.5 * (near_wfm_chopper["position"].data + 

59 far_wfm_chopper["position"].data) - data.meta["source_position"] 

60 # Ratio of WFM chopper distances 

61 z_ratio_wfm = ( 

62 sc.norm(far_wfm_chopper["position"].data - data.meta["source_position"]) / 

63 sc.norm(near_wfm_chopper["position"].data - data.meta["source_position"])) 

64 # Distance between detector positions and wfm chopper mid-point 

65 zdet_minus_zwfm = sc.norm(detector_positions - z_wfm) 

66 

67 # Neutron mass to Planck constant ratio 

68 alpha = sc.to_unit(constants.m_n / constants.h, 'us/m/angstrom') 

69 

70 # Frame time corrections: these are the mid-time point between the WFM choppers, 

71 # which is the same as the opening edge of the second WFM chopper in the case 

72 # of optically blind choppers. 

73 frames["time_correction"] = ch.time_open(far_wfm_chopper) 

74 

75 # Find delta_t for the min and max wavelengths: 

76 # dt_lambda_max is equal to the time width of the WFM choppers windows 

77 dt_lambda_max = ch.time_closed(near_wfm_chopper) - ch.time_open(near_wfm_chopper) 

78 

79 # t_lambda_max is found from the relation between t and delta_t: equation (2) in 

80 # Schmakat et al. (2020). 

81 t_lambda_max = (dt_lambda_max / dz_wfm) * zdet_minus_zwfm 

82 

83 # t_lambda_min is found from the relation between lambda_N and lambda_N+1, 

84 # equation (3) in Schmakat et al. (2020). 

85 t_lambda_min = t_lambda_max * z_ratio_wfm - data.meta["source_pulse_length"] * ( 

86 zdet_minus_zwfm / 

87 sc.norm(near_wfm_chopper["position"].data - data.meta["source_position"])) 

88 

89 # dt_lambda_min is found from the relation between t and delta_t: equation (2) 

90 # in Schmakat et al. (2020), and using the expression for t_lambda_max. 

91 dt_lambda_min = dt_lambda_max * z_ratio_wfm - data.meta[ 

92 "source_pulse_length"] * dz_wfm / sc.norm(near_wfm_chopper["position"].data - 

93 data.meta["source_position"]) 

94 

95 # Compute wavelength information 

96 lambda_min = t_lambda_min / (alpha * zdet_minus_zwfm) 

97 lambda_max = t_lambda_max / (alpha * zdet_minus_zwfm) 

98 dlambda_min = dz_wfm * lambda_min / zdet_minus_zwfm 

99 dlambda_max = dz_wfm * lambda_max / zdet_minus_zwfm 

100 

101 # Frame edges and resolutions for each pixel. 

102 # The frames do not stop at t_lambda_min and t_lambda_max, they also include the 

103 # fuzzy areas (delta_t) at the edges. 

104 frames["time_min"] = t_lambda_min - (0.5 * 

105 dt_lambda_min) + frames["time_correction"] 

106 frames["delta_time_min"] = dt_lambda_min 

107 

108 frames["time_max"] = t_lambda_max + (0.5 * 

109 dt_lambda_max) + frames["time_correction"] 

110 frames["delta_time_max"] = dt_lambda_max 

111 frames["wavelength_min"] = lambda_min 

112 frames["wavelength_max"] = lambda_max 

113 frames["delta_wavelength_min"] = dlambda_min 

114 frames["delta_wavelength_max"] = dlambda_max 

115 

116 frames["wfm_chopper_mid_point"] = 0.5 * (near_wfm_chopper["position"].data + 

117 far_wfm_chopper["position"].data) 

118 

119 return frames
